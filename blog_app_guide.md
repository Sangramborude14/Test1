## 2. 🗄️ Database Schemas (Mongoose)

### 👤 Auth Model (`models/User.js`)
Stores user profiles and credentials.
```javascript
const mongoose = require('mongoose');

const userSchema = new mongoose.Schema({
    username: { type: String, required: true, unique: true },
    password: { type: String, required: true },
    firstName: { type: String, required: true },
    lastName:  { type: String, required: true },
    gender:    { type: String, enum: ['Male', 'Female', 'Other'] },
    email:     { type: String, required: true, unique: true }
});

module.exports = mongoose.model('User', userSchema);
```

### 📝 Blog Model (`models/Blog.js`)
Stores blog posts and linked attachments.
```javascript
const mongoose = require('mongoose');

const blogSchema = new mongoose.Schema({
    title:       { type: String, required: true },
    blogContent: { type: String, required: true },
    timestamp:   { type: Date, default: Date.now },
    username:    { type: String, required: true }, // Links blog to creator
    filePath:    { type: String }                  // Storage path for files
});

module.exports = mongoose.model('Blog', blogSchema);
```

## 3. 🔐 Session Authentication Setup
In your `server.js`, configure the session store using MongoDB. This allows sessions to persist even after server restarts.

```javascript
const session = require('express-session');
const MongoStore = require('connect-mongo');

app.use(session({
    secret: 'mySecureSessionSecret',
    resave: false,
    saveUninitialized: false,
    store: MongoStore.create({ mongoUrl: 'mongodb://localhost:27017/blogDB' }),
    cookie: { 
        maxAge: 1000 * 60 * 60 * 24, // 24 Hours
        httpOnly: true 
    }
}));
```

> [!TIP]
> Always hash passwords using `bcrypt` during registration and verify them with `bcrypt.compare()` during login.

## 4. 📁 File Upload/Download (Multer)
Multer handles `multipart/form-data` for file uploads.

### Initializing Multer
```javascript
const multer = require('multer');
const path = require('path');

const storage = multer.diskStorage({
    destination: './uploads/',
    filename: (req, file, cb) => {
        cb(null, Date.now() + path.extname(file.originalname));
    }
});

const upload = multer({ storage });
```

### Download Endpoint
Use `res.download()` to serve files.
```javascript
// Download a specific file
router.get('/download/:filename', (req, res) => {
    const fileLoc = path.join(__dirname, '../uploads/', req.params.filename);
    res.download(fileLoc);
});
```
